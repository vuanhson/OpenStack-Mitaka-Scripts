- Cài đặt Neutron trên Compute node, sử dụng OpenvSwitch, mô hình provider network
- Cài đặt trên compute node

apt-get -y install neutron-server neutron-plugin-ml2 neutron-plugin-openvswitch-agent

- Cấu hình neutron trên compute node với các file như sau

############### FILE /etc/neutron/neutron.conf  ###############

[DEFAULT]
core_plugin = ml2
service_plugins = router
allow_overlapping_ips = True
auth_strategy = keystone
rpc_backend = rabbit


[agent]
root_helper = sudo /usr/bin/neutron-rootwrap /etc/neutron/rootwrap.conf

[cors]
[cors.subdomain]
[database]
# connection = mysql+pymysql://neutron:Welcome123@10.10.10.140/neutron

[keystone_authtoken]

user_domain_name = default

password = Welcome123

[matchmaker_redis]
[nova]

[oslo_messaging_notifications]

############### FILE /etc/neutron/plugins/ml2/ml2_conf.ini ###############

[DEFAULT]
[ml2]
type_drivers = flat,vlan,vxlan,gre
tenant_network_types = vlan,gre,vxlan
mechanism_drivers = openvswitch,l2population
extension_drivers = port_security
[ml2_type_flat]
flat_networks = external
[ml2_type_geneve]
[ml2_type_gre]
[ml2_type_vlan]
[ml2_type_vxlan]
[securitygroup]
